!
! L3 peering config between ${name} and ${neighbor}
!
default interface Eth ${my_port_num}
interface Eth ${my_port_num}
    description "L3 to ${neighbor} ${neighbor_port}"
    mtu ${mtu}
    no switchport access Vlan
    ip vrf forwarding Vrf_${vrf_name}
    no ip address
    ip address ${my_address}
    no shutdown
    no priority-flow-control priority
    no queue 3 wred-policy
    scheduler-policy ROCE
    qos-map dscp-tc ROCE
    qos-map dot1p-tc ROCE
    qos-map tc-queue ROCE
    qos-map tc-pg ROCE
    qos-map pfc-priority-queue ROCE
    qos-map pfc-priority-pg ROCE
    priority-flow-control priority 5,6
    priority-flow-control watchdog on detect-time 200
    priority-flow-control watchdog action drop
    priority-flow-control watchdog restore-time 400
    exit
!
