#
# L3 peering config between ${name} and ${neighbor}
#
interface ${my_port}
    default
    description L3 to ${neighbor}
    port link-mode route
    priority-flow-control enable
    priority-flow-control no-drop dot1p 5
    ip binding vpn-instance ${vrf_name}
    mtu ${mtu}
    no ip address
    ip address ${my_address_netmask}
    qos trust dscp
    qos wfq weight
    exit
