#
# L3 uplink peering config between ${name} and ${neighbor}
#
default interface ethernet ${my_port_num}
interface ethernet ${my_port_num}
    description L3 to ${neighbor}
    mtu ${mtu}
    no switchport force 
    vrf forwarding ${vrf_name}
    # access lists need to be added in the future
    no ip address
    ip address ${my_address} primary
    no shutdown
    exit
