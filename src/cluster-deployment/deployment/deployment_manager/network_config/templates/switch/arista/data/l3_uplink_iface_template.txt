!
! L3 uplink peering config between ${name} and ${neighbor}
!
default interface ${my_port}
interface ${my_port}
    description L3 to ${neighbor}
    mtu ${mtu}
    no switchport mode trunk
    no switchport access vlan
    no switchport
    vrf ${vrf_name}
    no ip address
    ip address ${my_address}
    ip access-group fw-allow-in in
    ip access-group fw-allow-out out
    no shutdown
!
ip access-list fw-allow-in
   5 permit tcp any any established
   10 permit tcp any host ${mgmt_node_data_address} eq https
   20 permit tcp any host ${mgmt_node_data_address} eq www
   30 permit tcp any host ${mgmt_node_data_address} established
   35 permit udp any eq rlzdbase any
   40 permit icmp any any
   50 permit tcp any host ${my_address} eq bgp
   60 deny ip any any log
!
ip access-list fw-allow-out
   10 permit tcp any any eq nfs
   20 permit tcp host ${fw_address} any eq bgp
   30 permit tcp host ${fw_address} eq bgp any
   40 permit udp any any eq rlzdbase
   50 permit icmp any any
   55 permit tcp any any established
   60 deny ip any any
!
